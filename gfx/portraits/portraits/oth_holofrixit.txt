portrait_groups = {
	oth_holofrixit = {
		default = holofrixit_male_01
		game_setup = { #will run with a limited country scope. species and government is set but the country does not actually exist
			add = {
				trigger = {
					ruler = { gender = male }
				}
				portraits = {
					holofrixit_male_01
					holofrixit_drone_01
					holofrixit_scientist_01
					holofrixit_warrior_01
					holofrixit_warrior_02
				}
			}
			add = {
				trigger = {
					ruler = { gender = female }
				}
				portraits = {
					holofrixit_female_01
					holofrixit_drone_01
					holofrixit_scientist_01
					holofrixit_warrior_01
					holofrixit_warrior_02
				}
			}

		}

		#species scope
		species = { #generic portrait for a species
			add = {
				portraits = {
					holofrixit_male_01
					holofrixit_female_01
				}
			}
		}

		#pop scope
		pop = { #for a specific pop
			add = {
				portraits = {
					holofrixit_drone_01
					holofrixit_drone_02
				}
			}
		}

		#leader scope
		leader = { #scientists, generals, admirals, governor
			add = {
				trigger = {
					gender = female
					NOT = { leader_class = scientist }
					NOT = { leader_class = general }
					NOT = { leader_class = admiral }
				}
				portraits = { holofrixit_female_01 }
			}
			add = {
				trigger = {
					gender = male
					NOT = { leader_class = scientist }
					NOT = { leader_class = general }
					NOT = { leader_class = admiral }
				}
				portraits = { holofrixit_male_01 }
			}
			add = {
				trigger = { gender = female leader_class = scientist }
				portraits = { holofrixit_scientist_01 }
			}
			add = {
				trigger = { gender = male leader_class = scientist }
				portraits = { holofrixit_scientist_01 }
			}
			add = {
				trigger = { gender = female leader_class = general }
				portraits = { holofrixit_warrior_01 }
			}
			add = {
				trigger = { gender = male leader_class = general }
				portraits = { holofrixit_warrior_01 }
			}
			add = {
				trigger = { gender = female leader_class = admiral }
				portraits = { holofrixit_scientist_01 }
			}
			add = {
				trigger = { gender = male leader_class = admiral }
				portraits = { holofrixit_scientist_01 }
			}
			add = {
				trigger = { gender = female leader_class = governor }
				portraits = { holofrixit_female_01 }
			}
			add = {
				trigger = { gender = male leader_class = governor }
				portraits = { holofrixit_male_01 }
			}
		}


		#leader scope
		ruler = {
			add = {
				trigger = { gender = female }
				portraits = { holofrixit_female_01 }
			}
			add = {
				trigger = { gender = male }
				portraits = { holofrixit_male_01 }
			}
		}
	}
}
