		}

	}

	#leader scope
	leader = { #rulers, rulers, admirals, ruler
		"gfx/models/portraits/r27/hum_white/whitewalker_f_outfit_ruler.dds" = { leader_class = ruler }
		"gfx/models/portraits/r27/hum_white/whitewalker_f_outfit_ruler.dds" = { leader_class = ruler }
		"gfx/models/portraits/r27/hum_white/whitewalker_f_outfit_ruler.dds" = { leader_class = admiral }
		"gfx/models/portraits/r27/hum_white/whitewalker_f_outfit_ruler.dds" = { leader_class = ruler }
	}

	#leader scope
	ruler = { #for rulers
		default = "gfx/models/portraits/r27/hum_white/whitewalker_f_outfit_ruler.dds"

		"gfx/models/portraits/r27/hum_white/whitewalker_f_outfit_ruler.dds" = { owner = { has_generic_government = yes } }
		"gfx/models/portraits/r27/hum_white/whitewalker_f_outfit_ruler.dds" = { owner = { has_spiritualist_government = yes } }
		"gfx/models/portraits/r27/hum_white/whitewalker_f_outfit_ruler.dds" = { owner = { has_militarist_government = yes } }
		"gfx/models/portraits/r27/hum_white/whitewalker_f_outfit_ruler.dds" = { owner = { has_pacifist_government = yes } }
		"gfx/models/portraits/r27/hum_white/whitewalker_f_outfit_ruler.dds" = { owner = { has_primitive_government = yes } }
		"gfx/models/portraits/r27/hum_white/whitewalker_f_outfit_ruler.dds" = { owner = { has_materialist_government = yes } }
	}
}
